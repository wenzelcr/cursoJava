//Faça um programa em Java que calcula e escreve a seguinte soma: soma = 1/1 + 3/2 + 5/3 + 7/4 + ... + 99/50

import java.util.Scanner;

public class MeuPrograma {

    public static void main(String[] args){

        Scanner scanner = new Scanner(System.in);
        double divisor = 1;
        double valor;
        double soma = 0;

        System.out.print("Digite o valor para verificação se é perfeito: ");
        valor = scanner.nextDouble();

        while (divisor < valor){

            if( (valor % divisor) == 0 ){
                soma = soma + divisor;
            }
            divisor++;
        }
        if (valor==soma){
            System.out.println( "O numero " + valor + " é perfeito!");
        }else{
            System.out.println( "O numero " + valor + " NÃO é perfeito!");
        }
    }
}
